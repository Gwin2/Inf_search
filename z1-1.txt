Лабораторная работа N 1
Язык MySQL

1. Все покупатели, для которых номер продавца равен 1001:
SELECT *
FROM cust
WHERE snum = 1001;

2. Таблица продавцов со столбцами city, sname, snum, comm:
SELECT city, sname, snum, comm
FROM sal;

3. Оценка и имя каждого покупателя в городе San Jose:
SELECT rating, cname
FROM cust
WHERE city = 'San Jose';

4. Номера продавцов из таблицы заказов без повторений:
SELECT DISTINCT snum
FROM ord;

5. Поля sname и city для всех продавцов в London с комиссионными строго больше 0.11:
SELECT sname, city
FROM sal
WHERE city = 'London'
  AND comm > 0.11;

6. Все покупатели с рейтингом меньше или равным 200, если они не находятся в Rome:
SELECT *
FROM cust
WHERE rating <= 200
  AND city <> 'Rome';

7. Все заказы на 3 и 5 октября 1990 г. Способ 1:
SELECT *
FROM ord
WHERE odate IN ('1990-10-03', '1990-10-05');

   Способ 2:
SELECT *
FROM ord
WHERE odate = '1990-10-03'
   OR odate = '1990-10-05';

8. Все покупатели, чьи имена начинаются с буквы от A до G:
SELECT *
FROM cust
WHERE cname REGEXP '^[A-G]';

9. Все продавцы, имена которых содержат букву e:
SELECT *
FROM sal
WHERE sname LIKE '%e%';

10. Сумма всех заказов на 3 октября 1990 г.:
SELECT SUM(amt) AS total_amt
FROM ord
WHERE odate = '1990-10-03';

11. Сумма всех заказов для продавца с номером 1001:
SELECT SUM(amt) AS total_amt
FROM ord
WHERE snum = 1001;

12. Наибольший заказ для каждого продавца:
SELECT snum, MAX(amt) AS max_amt
FROM ord
GROUP BY snum;

13. Покупатель, чье имя первое по алфавиту среди имен, заканчивающихся на s:
SELECT *
FROM cust
WHERE cname LIKE '%s'
ORDER BY cname
LIMIT 1;

14. Средние комиссионные в каждом городе:
SELECT city, AVG(comm) AS avg_comm
FROM sal
GROUP BY city;

15. Для каждого заказа на 3 октября: номер, стоимость в евро, имя продавца и комиссионные продавца за заказ:
SELECT o.onum,
       o.amt * 0.8 AS amt_euro,
       s.sname,
       o.amt * s.comm AS seller_comm
FROM ord AS o
JOIN sal AS s ON o.snum = s.snum
WHERE o.odate = '1990-10-03';

16. Номера заказов по возрастанию, имена продавцов и покупателей, если продавец находится в London или Rome:
SELECT o.onum, s.sname, c.cname
FROM ord AS o
JOIN sal AS s ON o.snum = s.snum
JOIN cust AS c ON o.cnum = c.cnum
WHERE s.city IN ('London', 'Rome')
ORDER BY o.onum;

17. Имена продавцов по алфавиту, суммы заказов до 5 октября и полученные комиссионные:
SELECT s.sname,
       SUM(o.amt) AS total_amt,
       SUM(o.amt * s.comm) AS total_comm
FROM sal AS s
JOIN ord AS o ON s.snum = o.snum
WHERE o.odate < '1990-10-05'
GROUP BY s.snum, s.sname
ORDER BY s.sname;

18. Номера заказов, стоимость, имена продавцов и покупателей, если города продавцов и покупателей начинаются с букв от L до R:
SELECT o.onum, o.amt, s.sname, c.cname
FROM ord AS o
JOIN sal AS s ON o.snum = s.snum
JOIN cust AS c ON o.cnum = c.cnum
WHERE s.city REGEXP '^[L-R]'
  AND c.city REGEXP '^[L-R]';

19. Все пары покупателей, обслуживаемые одним продавцом, без самих себя и обратных пар:
SELECT c1.cname AS cname1,
       c2.cname AS cname2,
       c1.snum
FROM cust AS c1
JOIN cust AS c2 ON c1.snum = c2.snum
WHERE c1.cnum < c2.cnum;

20. Имена всех покупателей, чьи продавцы имеют комиссионные меньше 0.13:
SELECT cname
FROM cust
WHERE snum IN (
    SELECT snum
    FROM sal
    WHERE comm < 0.13
);

21. Создание копии таблицы продавцов с данными из SAMPLE.SAL, проверка структуры и данных:
CREATE TABLE sal_copy AS
SELECT *
FROM sample.sal;

DESC sample.sal;
DESC sal_copy;

SELECT *
FROM sample.sal
ORDER BY snum;

SELECT *
FROM sal_copy
ORDER BY snum;

SELECT 'only_in_original' AS diff_type, s.*
FROM sample.sal AS s
LEFT JOIN sal_copy AS c
       ON s.snum = c.snum
      AND s.sname = c.sname
      AND s.city = c.city
      AND s.comm = c.comm
WHERE c.snum IS NULL
UNION ALL
SELECT 'only_in_copy' AS diff_type, c.*
FROM sal_copy AS c
LEFT JOIN sample.sal AS s
       ON s.snum = c.snum
      AND s.sname = c.sname
      AND s.city = c.city
      AND s.comm = c.comm
WHERE s.snum IS NULL;

22. Вставка двух новых записей, вывод таблицы, удаление одной новой записи, повторный вывод:
INSERT INTO sal_copy (snum, sname, city, comm)
VALUES (2001, 'Ivanov', 'Moscow', 0.10),
       (2002, 'Petrov', 'Kazan', 0.12);

SELECT *
FROM sal_copy;

DELETE FROM sal_copy
WHERE snum = 2001;

SELECT *
FROM sal_copy;

23. Вставка двух строк, увеличение комиссионных в 2 раза, вывод после каждого изменения:
INSERT INTO sal_copy (snum, sname, city, comm)
VALUES (2003, 'Sidorov', 'Perm', 0.09),
       (2004, 'Smirnov', 'Omsk', 0.11);

SELECT *
FROM sal_copy;

UPDATE sal_copy
SET comm = comm * 2;

SELECT *
FROM sal_copy;
